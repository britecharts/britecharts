Brush chart to use with other charts as a time range selector.

The code


brushChart
    .width(containerWidth)
    .height(300)
    .on('customBrushStart', function(brushExtent) {
        // Do something with the brushExtent
    });

brushContainer.datum(dataset).call(brushChart);
