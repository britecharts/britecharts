Legend with Donut Chart

Legend chart to use with Donut charts to show definitions of color indicators, text labels, and their corresponding values.

The code


donutChart
    .on('customMouseOver', function(data) {
        legendChart.highlight(data.data.id);
    })
    .on('customMouseOut', function() {
        legendChart.clearHighlight();
    });

legendChart
    .height(legendContainerWidth / 3)
    .width(legendContainerWidth / 2)
    .numberFormat('s');

donutContainer.datum(dataset).call(donutChart);
legendContainer.datum(dataset).call(legendChart);

Small Donut Chart with Inline Legend

The code


donutChart
    .on('customMouseOver', function(data) {
        legendChart.highlight(data.data.id);
    })
    .on('customMouseOut', function() {
        legendChart.clearHighlight();
    });

legendChart
    .isHorizontal(true)
    .markerSize(8)
    .height(40)
    .width(legendContainerWidth*0.6);

donutContainer.datum(dataset).call(donutChart);
legendContainer.datum(dataset).call(legendChart);
